DROP TABLE IF EXISTS `spdxhub`.`spdx_docs`;
CREATE TABLE `spdxhub`.`spdx_docs` (
`id` int(11) NOT NULL AUTO_INCREMENT,
`spdx_version` varchar(255) DEFAULT NULL,
`data_license` varchar(255) DEFAULT NULL,
`upload_file_name` varchar(255) DEFAULT NULL,
`upload_content_type` varchar(255) DEFAULT NULL,
`upload_file_size` varchar(255) DEFAULT NULL,
`upload_updated_at` datetime DEFAULT NULL,
`document_comment` text DEFAULT NULL,
`created_at` datetime NOT NULL,
`updated_at` datetime NOT NULL,
PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=55 DEFAULT CHARSET=latin1;
